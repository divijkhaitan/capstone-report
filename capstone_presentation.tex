    \begin{frame}{Evaluating Neural Networks}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \item 
                Deep Learning Models are usually trained and evaluated on a single dataset
            \item 
                Accuracy and related metrics only defined on a single dataset, no claims made about correctness of an output
            \item 
                Handling Changes to input data?
        \end{itemize}
    \end{frame}

    \begin{frame}{Shifts in Data Distributions: Examples}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \item 
                Bank Fraud: Training on data from rural areas and applying it to urban areas
            \item 
                Medical Imaging: Cancer classification model for European or Middle Eastern populations used in subcontinental hosptials
            \item 
                Autonomus Vehicles: Navigating mountain terrain after training for cities and paved roads
        \end{itemize}
    \end{frame}

    \begin{frame}{Similarity of Data}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \item 
                In all of the above applications, neither is the universal set clear nor the probability distribution
            \item 
                Metrics in a high dimensional space are not meaningful assuming the manifold hypothesis
            \item 
                Distributions/Metrics in a denoised embedding space may be better to focus on
        \end{itemize}
    \end{frame}

    \begin{frame}{Manifold Hypothesis}
        \includegraphics[width=\textwidth]{manifold.png}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \setlength\itemsep{1em}
            \item 
                Pre-dates deep learning
            \item 
                Data consists of substructures with much smaller dimension than the ambient space
            \item 
                Valuable as a heuristic, basis of most dimensionality reduction algorithms
        \end{itemize}
    \end{frame}

    \begin{frame}{Circuits and Superposition}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \item 
                First studied in \cite{olah2020zoom}, which argues that alongside the directions of the feature space the interactions between them are important
            \item 
                Find that there are circuit structures, particularly in image detections where earlier layer detect lower level features like edges and later layers detect complex features like shapes and faces
            \item 
                Concepts encoded by monosemantic neurons in earlier layers may be encoded by polysemantic neurons in subsequent 
            \item 
                Coined the term superposition which is the distribution multiple exclusive featues across the same set of directions
        \end{itemize}
    \end{frame}

    \begin{frame}{Granularities to be Studied}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \item 
                Circuits/Neuron-Weight Interactions as bernoulli trials: Stable fits in exchange for losing higher moments of distributions
            \item 
                Circuits/Neuron-Weight Interactions as multivariate gaussians: Computationally very intensive to model, unstable fits but captures second moment
            \item 
                Neurons: Captures second moment of the distribution over activations but lose out on path information, computationally more tractable than gaussians over circuits
        \end{itemize}
    \end{frame}

    \begin{frame}{Neuron-Weight Interactions}
        Every product of a neuron from the previous layer with a weight from the next layer. 
        Given an activation vector $x \in \mathbb{R}^{n_{in} \times 1}$ and a weight matrix $W \in \mathbb{R}^{n_{out} \times n_{in}}$ 
        \[
            \begin{bmatrix}
                w_{1,1} & \dots & w_{1,n_{in}} \\
                \vdots & \ddots & \vdots \\
                w_{n_{out}, 1} & \cdots & w_{n_{out}, n_{in}} \\
            \end{bmatrix}
            \begin{bmatrix}
                x_1 \\
                \vdots \\
                x_{n_{in}} \\
            \end{bmatrix}
            = 
            \begin{bmatrix}
                \sum_{i=1}^{n_{in}} w_{1,i}x_i\\
                \vdots \\
                \sum_{i=1}^{n_{in}} w_{n_{out},i}x_i\\
            \end{bmatrix}
        \]
    \end{frame}
    
    \begin{frame}{Modelling Circuits/Neuron-Weight Interactions}
        To study every term in each summation individually, a vector of 1s needs to be factored out.
        \[
            W\diag(x_1, \dots, x_n)    
            = 
            \begin{bmatrix}
                w_{1, 1}x_1 & \dots & w_{1, n_{in}}x_{n_{in}} \\
                \vdots & \ddots & \vdots \\
                w_{n_{out}, 1}x_1 & \dots & w_{n_{out}, n_{in}}x_{n_{in}}
            \end{bmatrix}
        \]
    \end{frame}
    
    \begin{frame}{Modelling Circuits/Neuron-Weight Interactions: Bernoulli}
        An interaction is considered significant if it is large relative to the magnitude of the neuron value, corrected for the input dimension. These are flattened into a single vector for which the success proportion for each coordinate $\hat{p}$ is estimated using the MLE $\hat{p}_{i, j} = \frac{1}{k}\sum {X_{i, j}^l}$, where $l$ indexes over the $k$ datapoints
        \[X = 
        \begin{bmatrix}
            |w_{1,1}x_1| > \frac{\sum_j w_{1j}x_j}{n_{in}} \\
            \vdots \\
            |w_{1,n_{in}}x_{n_{in}}| > \frac{\sum_j w_{1j}x_j}{n_{in}}\\
            \vdots \\
            |w_{n_{out},n_{in}}x_{n_{in}}| > \frac{\sum_j w_{n_{out}j}x_j}{n_{in}}\\
        \end{bmatrix} \rightarrow 
        \begin{bmatrix}
            \hat{p}_{1,1} \\
            \vdots \\
            \hat{p}_{1,n_{in}} \\
            \vdots \\
            \hat{p}_{n_{out},n_{in}} \\
        \end{bmatrix} 
        \]
    \end{frame}

    \begin{frame}{Modelling Circuit Activations: Multivariate Gaussian}
        The matrix itself can also be flattened 
        \[X = 
        \begin{bmatrix}
            w_{1,1}x_1 \\
            \vdots \\
            w_{1,n_{in}}x_{n_{in}} \\
            \vdots \\
            w_{n_{out},n_{in}}x_{n_{in}} \\
        \end{bmatrix}
        \]
        From this we can estimate the mean and variance using the MLEs $\hat{\mu} = \frac{1}{k} \sum X^l$ and $\hat{\Sigma} = \frac{1}{k} \sum (X^l - \hat{\mu})(X^l - \hat{\mu})^T$
    \end{frame}

    \begin{frame}{Modelling Neurons: Multivariate Gaussian}
        The neurons can be written as  
        \[X = 
        \begin{bmatrix}
            x_1 \\
            \vdots \\
            x_{n_{in}} \\
        \end{bmatrix}
        \]
        Using the same MLEs as before, a multivariate gaussian can be fit to them as well
    \end{frame}

    \begin{frame}{Rationale and Limitations of the Multivariate Gaussian Model}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \item 
                Easy to fit, needs just mean and variance
            \item 
                Probabilities assuming parameters are easy to compute 
            \item 
                Ignores moments beyond variance (assumes symmetry/ignores skew)
            \item 
                Estimating the joint density of the kernel/a 2D histogram may capture the distributions with greater accuracy
        \end{itemize}
    \end{frame}

    \begin{frame}{Sparsity of Neuron Activations}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \setlength\itemsep{1em}
            \item 
                Practically, pruning techniques can drastically decrease the size of neural networks
            \item 
                Indicates that a small part of the network is important
            \item 
                Lottery Ticket Hypothesis and related work focuses on identiying sparse structures early on in training
            \item 
                Activations should be sparse
        \end{itemize}
    \end{frame}


    \begin{frame}{KL Divergence and Entropy}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \setlength\itemsep{1em}
            \item 
                Entropy of a distribution is a measure of the amount of information encoded in a random variable
                \[H(X) = -\sum_x p(x)\log(p(x))\]
            \item 
                KL divergence is the relative entropy between 2 variables, i.e. the amount of additional information needed to transform an encoding of P to one of Q
                \[KL(P||Q) = \sum_x P(x)\log(\frac{P(x)}{Q(x)})\]
            \item 
                KL divergence isn't a metric, satisfies nonnegavity but not symmetry or triangle inequality
        \end{itemize}
    \end{frame}

    \begin{frame}{Mahalanobis Distance}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \item 
                This is a notion of distance between a point and a distribution
            \item 
                Takes into account the variance structure of a distribution
            \item 
                For a multivariate gaussian distribution, it has the following closed form given a distribution $\mathcal{D}$ and a point $c$
                \[d_{\text{mahalanobis}}(\mathcal{D}, c) = \sqrt{((\vec{\mu_{\mathcal{D}}} - \vec{c})^T\Sigma_{\mathcal{D}}(\vec{\mu_\mathcal{D}} -\vec{c}))}\]
            \item 
                To find this between 2 distributions, we computed the distance of the mean of a target distribution from a base distribution
        \end{itemize}
    \end{frame}

    \begin{frame}{Activation Distributions}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \setlength\itemsep{1em}
            \item 
                Measured KL divergence and entropy for distributions
            \item 
                Can be seen as analogues to the cluster radius and distances
            \item 
                Our current methodology over-estimates the separation between classes
            \item 
                Large KL divergence $\rightarrow$ separated paths/neurons $\rightarrow$ robustness
        \end{itemize}
    \end{frame}

    \begin{frame}{Estimating Dimensions of a Manifold}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \setlength\itemsep{1em}
            \item 
                Extracted the output of the final hidden layer (embedding), and computed the dimensionality of the manifolds for each class
            \item 
                Final layer activations must be linearly separable 
            \item 
                Computed using principal components upto 95\% explained variance
            \item 
                Measured pairwise entangled classes by computing the rank of the matrix created by the union of PCs
        \end{itemize}
    \end{frame}

    \begin{frame}{Overview of Framework}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \item 
                Distributions over neuron-weight interactions: Increase in the separation between classes $\implies$ developing a distinct idea for each class
            \item 
                Sparse Activations: Small number of frequent firings, large number of random firings
            \item 
                OOD Data: Natural probabilistic definition of out-of-distribution data rather than geometric definition as with manifolds
        \end{itemize}
    \end{frame}

    \begin{frame}{Conclusions}
        \begin{itemize}
            \setlength{\itemsep}{2em}
            \setlength\itemsep{1em}
            \item 
                Significant activations are sparse
            \item 
                Framework registers different behaviour on in and out of distribution datasets and memorising and best-fit networks
            \item 
                PGD reduces norms in the embedding space
            \item 
                Dataset level shifts are difficult to detect
            \item 
                Semantics learnt by a network are different than our understanding
        \end{itemize}
    \end{frame}
